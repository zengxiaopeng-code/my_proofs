\chapter{Lemma 1: Posterior process}

This chapter formalizes \textbf{Lemma 1 (Posterior process)} of the paper (label
\texttt{lem:posterior-martingale}). \textbf{The verbatim statement and proof are in
\texttt{docs/paper-lemma1.md}}; this chapter decomposes that proof, following the paper's own
four steps, into separate nodes. Each node links to its Lean declaration and is marked green
exactly when it is fully verified by the Lean kernel (no \texttt{sorry}). Click any node to
check what it states and whether Lean has actually proved it.

\begin{definition}[Environment]\label{def:setup}
    Discrete time $t\in\mathbb T=\{0,1,2,\dots\}$. A Borel probability space
    $(\Omega,\mathcal F,\mathbb P)$; primitive state $\theta:\Omega\to\Theta$ with $\Theta$
    compact Polish; $\Delta(\Theta)$ the space of Borel probability measures on $\Theta$ (weak
    topology). Public state $Z_t:\Omega\to E_t$ ($E_t$ Polish), with $\mathcal F_t=\sigma(Z_t)$
    an increasing filtration and $\mathcal F_0$ trivial.
\end{definition}

%======================================================================
% Main statement (verbatim); its proof is decomposed into the four step-nodes below.
%======================================================================
\begin{lemma}[Posterior process]\label{lem:posterior-martingale}
    \uses{def:setup}
    For each $t\in\mathbb T$, there exists an $\mathcal F_t$-measurable posterior kernel
    $S_t:\Omega\to\Delta(\Theta)$ such that, for every bounded Borel function
    $\varphi:\Theta\to\mathbb R$,
    \[
      \int_\Theta \varphi(\theta')\,S_t(\omega)(d\theta')
        = \mathbb E[\varphi(\theta)\mid\mathcal F_t](\omega)\qquad\text{a.s.}
    \]
    In particular:
    \textbf{(i)} $S_t$ is a sufficient statistic for $\theta$ given $\mathcal F_t$;
    \textbf{(ii)} since $\mathcal F_t = \sigma(Z_t)$, the Doob--Dynkin lemma implies that
    $S_t = f_t(Z_t)$ for some Borel measurable $f_t$;
    \textbf{(iii)} $\{S_t\}_{t\in\mathbb T}$ is a bounded $(\mathcal F_t)$-martingale.
\end{lemma}
\begin{proof}
    \uses{lem:existence, lem:sufficiency, lem:factorization, lem:martingale}
    The statement is the conjunction of the four steps below: existence and the kernel property
    (Lemma~\ref{lem:existence}), sufficiency (Lemma~\ref{lem:sufficiency}, part~(i)),
    factorization through $Z_t$ (Lemma~\ref{lem:factorization}, part~(ii)), and the martingale
    property (Lemma~\ref{lem:martingale}, part~(iii)). The verbatim proof is in
    \texttt{docs/paper-lemma1.md}.
\end{proof}

%======================================================================
% Step 1: existence + kernel property
%======================================================================
\begin{lemma}[Existence of the posterior kernel]\label{lem:existence}
    \lean{PosteriorProcess.kernel_exists}
    \uses{def:setup}
    For each $t$, there exists an $\mathcal F_t$-measurable $S_t:\Omega\to\Delta(\Theta)$ such
    that, for every bounded Borel $\varphi$, the map
    $\omega\mapsto\int_\Theta\varphi\,dS_t(\omega)$ equals
    $\mathbb E[\varphi(\theta)\mid\mathcal F_t]$ almost surely.
\end{lemma}
\begin{proof}
    $\Theta$ compact Polish $\Rightarrow$ standard Borel; the claim follows from the existence
    of regular conditional distributions (Kallenberg 2002, Thm~6.3).
    \emph{Lean status:} declared as \texttt{PosteriorProcess.kernel\_exists}; the proof still
    contains \texttt{sorry} (not yet verified).
\end{proof}

%======================================================================
% Step 2: (i) sufficiency
%======================================================================
\begin{lemma}[Part (i): Sufficiency]\label{lem:sufficiency}
    \uses{lem:existence}
    Conditioning on $S_t$ and conditioning on $\mathcal F_t$ yield the same conditional
    distribution of $\theta$; that is, $S_t$ is a sufficient statistic for $\theta$ given
    $\mathcal F_t$.
\end{lemma}
\begin{proof}
    For bounded Borel $\varphi$, the map $F(\mu):=\int\varphi\,d\mu$ is Borel measurable on
    $\Delta(\Theta)$ by the functional monotone class theorem (it holds for continuous $\varphi$
    by the definition of the weak topology, and extends to bounded Borel $\varphi$ by dominated
    convergence). Hence $\mathbb E[\varphi(\theta)\mid\mathcal F_t]=F(S_t)$ is
    $\sigma(S_t)$-measurable, and uniqueness of conditional expectation gives the equality of the
    two conditional expectations. \emph{Lean status:} not yet formalized (no declaration).
\end{proof}

%======================================================================
% Step 3: (ii) Doob-Dynkin -- abstract layer (proved) + library gap + full version
%======================================================================
\begin{lemma}[Abstract Doob--Dynkin factorization]\label{lem:doob-abstract}
    \lean{PosteriorProcess.factorization}\leanok
    If $S$ is $\sigma(Z)$-measurable and takes values in a standard Borel space, then
    $S=f\circ Z$ for some Borel measurable $f$.
\end{lemma}
\begin{proof}\leanok
    Mathlib's \texttt{Measurable.exists\_eq\_measurable\_comp}. Verified by the Lean kernel, no
    \texttt{sorry}.
\end{proof}

\begin{lemma}[$\Delta(\Theta)$ is standard Borel]\label{lem:deltaTheta-sb}
    \lean{PosteriorProcess.deltaTheta_standardBorel}
    \uses{def:setup}
    $\Delta(\Theta)$ (Borel probability measures on the compact Polish space $\Theta$, weak
    topology) is a standard Borel space.
\end{lemma}
\begin{proof}
    True mathematically (the space of probability measures on a compact Polish space is again
    compact Polish). \emph{Lean status:} Mathlib currently lacks the
    \texttt{PolishSpace}/\texttt{StandardBorelSpace} instance for \texttt{ProbabilityMeasure}, so
    \texttt{PosteriorProcess.deltaTheta\_standardBorel} contains \texttt{sorry} (library gap).
\end{proof}

\begin{lemma}[Part (ii): Induction by $Z_t$]\label{lem:factorization}
    \lean{PosteriorProcess.factorization_posterior}
    \uses{lem:existence, lem:doob-abstract, lem:deltaTheta-sb}
    Since $\mathcal F_t=\sigma(Z_t)$, there is a Borel measurable $f_t:E_t\to\Delta(\Theta)$ with
    $S_t=f_t(Z_t)$ a.s.; the public history $Z_t$ induces the posterior $S_t$.
\end{lemma}
\begin{proof}
    Apply the abstract factorization (Lemma~\ref{lem:doob-abstract}) to $\Delta(\Theta)$, which
    requires it to be standard Borel (Lemma~\ref{lem:deltaTheta-sb}). \emph{Lean status:}
    declared as \texttt{factorization\_posterior}; the abstract step is verified, but the whole
    still depends on the library gap of Lemma~\ref{lem:deltaTheta-sb} and thus contains
    \texttt{sorry}.
\end{proof}

%======================================================================
% Step 4: (iii) martingale property (proved)
%======================================================================
\begin{lemma}[Part (iii): Martingale property]\label{lem:martingale}
    \lean{PosteriorProcess.testfun_martingale}\leanok
    \uses{lem:existence}
    For every bounded Borel $\varphi$, the real-valued process
    $t\mapsto\int_\Theta\varphi\,dS_t$ is an $(\mathcal F_t)$-martingale; hence $\{S_t\}$ is a
    bounded measure-valued martingale.
\end{lemma}
\begin{proof}\leanok
    By the tower property, $\mathbb E[\int\varphi\,dS_t\mid\mathcal F_s]
    =\mathbb E[\varphi(\theta)\mid\mathcal F_s]=\int\varphi\,dS_s$ for $s\le t$. Mathlib's
    \texttt{martingale\_condExp}. Verified by the Lean kernel, no \texttt{sorry}.
\end{proof}
