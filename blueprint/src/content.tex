\begin{definition}[Setup]\label{def:setup}
    概率空间 $(\Omega,\mathcal F,\mathbb P)$，紧 Polish（故标准 Borel）状态空间
    $\Theta$，滤子 $(\mathcal F_t)_{t\in\mathbb T}$ 满足 $\mathcal F_t=\sigma(Z_t)$，
    随机变量 $\theta:\Omega\to\Theta$，公开历史 $Z_t$。
\end{definition}

\begin{lemma}[Posterior kernel: existence]\label{lem:existence}
    \lean{PosteriorProcess.kernel_exists}
    \uses{def:setup}
    对每个 $t$，存在 $\mathcal F_t$-可测的后验核 $S_t:\Omega\to\Delta(\Theta)$，
    使得对每个有界 Borel $\varphi$，
    $\int_\Theta\varphi\,dS_t=\mathbb E[\varphi(\theta)\mid\mathcal F_t]$ a.s.
\end{lemma}
\begin{proof}
    \uses{def:setup}
    $\Theta$ 标准 Borel，由正则条件分布定理（Kallenberg Thm 6.3）即得。
    % \leanok  <- 证完 Lean 后取消注释
\end{proof}

\begin{lemma}[Part (i): sufficiency]\label{lem:sufficiency}
    \uses{lem:existence}
    $S_t$ 是 $\theta$（给定 $\mathcal F_t$）的充分统计量：对 $S_t$ 取条件与对
    $\mathcal F_t$ 取条件给出 $\theta$ 相同的条件分布。
\end{lemma}
\begin{proof}
    \uses{lem:existence}
    $F(\mu):=\int\varphi\,d\mu$ 在 $\Delta(\Theta)$ 上 Borel 可测（泛函单调类定理）。
    于是 $\mathbb E[\varphi(\theta)\mid\mathcal F_t]=F(S_t)$ 是 $\sigma(S_t)$-可测；
    再由条件期望唯一性得两条件期望相等。
    \emph{注：论文原脚注"有界 Borel = 连续函数逐点极限"不成立（Baire class），
    应删除，单调类论证已足够。}
\end{proof}

\begin{lemma}[Part (ii): factorization]\label{lem:doobdynkin}
    \lean{PosteriorProcess.factorization}
    \uses{lem:existence}
    存在 Borel 可测 $f_t$ 使 $S_t=f_t(Z_t)$ a.s.
\end{lemma}
\begin{proof}
    \uses{lem:existence}
    $S_t$ 是 $\sigma(Z_t)$-可测、取值标准 Borel 空间 $\Delta(\Theta)$，
    Doob--Dynkin 引理即得。
\end{proof}

\begin{lemma}[Part (iii): martingale]\label{lem:martingale}
    \lean{PosteriorProcess.testfun_martingale}
