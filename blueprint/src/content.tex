\chapter{Lemma 1: Posterior process}

本章形式化论文的 \textbf{Lemma 1 (Posterior process)}（论文标签
\texttt{lem:posterior-martingale}）。节点与论文一一对应；读者可直接按标题对回原文。

\begin{definition}[Environment]\label{def:setup}
    离散时间 $t\in\mathbb T=\{0,1,2,\dots\}$。Borel 概率空间 $(\Omega,\mathcal F,\mathbb P)$，
    原始状态 $\theta:\Omega\to\Theta$，$\Theta$ 紧 Polish；$\Delta(\Theta)$ 为 $\Theta$ 上概率测度
    （弱拓扑）。公开状态 $Z_t:\Omega\to E_t$（$E_t$ Polish），$\mathcal F_t=\sigma(Z_t)$ 为递增滤子，
    $\mathcal F_0$ 平凡。
\end{definition}

\begin{lemma}[Posterior process]\label{lem:posterior-martingale}
    \lean{PosteriorProcess.kernel_exists, PosteriorProcess.factorization_posterior,
      PosteriorProcess.testfun_martingale}
    \uses{def:setup}
    For each $t\in\mathbb T$, there exists an $\mathcal F_t$-measurable posterior kernel
    $S_t:\Omega\to\Delta(\Theta)$ such that, for every bounded Borel function
    $\varphi:\Theta\to\mathbb R$,
    \[
      \int_\Theta \varphi(\theta')\,S_t(\omega)(d\theta')
        = \mathbb E[\varphi(\theta)\mid\mathcal F_t](\omega)\qquad\text{a.s.}
    \]
    In particular:

    \textbf{(i)} $S_t$ is a sufficient statistic for $\theta$ given $\mathcal F_t$:
    conditioning on $S_t$ and conditioning on $\mathcal F_t$ yield the same conditional
    distribution of $\theta$.

    \textbf{(ii)} Since $\mathcal F_t = \sigma(Z_t)$, the Doob--Dynkin lemma implies that
    $S_t = f_t(Z_t)$ for some Borel measurable function $f_t$; that is, the public history
    $Z_t$ induces the posterior $S_t$.

    \textbf{(iii)} $\{S_t\}_{t\in\mathbb T}$ is a bounded $(\mathcal F_t)$-martingale.
\end{lemma}
% 证明正文 = 论文逐字原文（英文，未改）。权威档案见 docs/paper-lemma1.md；
% 每部分对应的 Lean 声明与验证状态，另见证明后的“形式化状态”段与 CORRESPONDENCE.md。
\begin{proof}
    \uses{def:setup}
    \emph{Existence of the posterior kernel.}\quad
    Since $\Theta$ is a compact Polish space, $\Delta(\Theta)$ equipped with the weak topology
    is also a compact Polish space, hence a standard Borel space. The map $\theta:\Omega\to\Theta$
    is $\mathcal F$-measurable and $\mathcal F_t\subseteq\mathcal F$ is a sub-$\sigma$-algebra. By
    the standard theorem on regular conditional distributions for Polish-space-valued random
    variables (Kallenberg 2002, Theorem~6.3), there exists a Markov kernel
    $S_t:\Omega\times\mathcal B(\Theta)\to[0,1]$ such that \textbf{(a)}
    $\omega\mapsto S_t(\omega)(B)$ is $\mathcal F_t$-measurable for every $B\in\mathcal B(\Theta)$,
    and \textbf{(b)} $S_t(\omega)(\cdot)=\mathbb P(\theta\in\cdot\mid\mathcal F_t)(\omega)$
    $\mathbb P$-a.s. Equivalently, for every bounded Borel $\varphi:\Theta\to\mathbb R$,
    \[
      \int_\Theta\varphi(\theta')\,S_t(\omega)(d\theta')
        =\mathbb E[\varphi(\theta)\mid\mathcal F_t](\omega)\qquad\text{a.s.}
        \qquad(\star)
    \]

    \emph{Part (i): Sufficiency.}\quad
    Fix any bounded Borel $\varphi:\Theta\to\mathbb R$ and define $F:\Delta(\Theta)\to\mathbb R$
    by $F(\mu):=\int_\Theta\varphi(\theta')\,\mu(d\theta')$. By the functional monotone class
    theorem, $F$ is Borel measurable on $\Delta(\Theta)$: it holds for continuous $\varphi$ by
    the definition of the weak topology, and extends to all bounded Borel $\varphi$ by dominated
    convergence.\footnote{Explicitly: every bounded Borel function is the pointwise limit of a
    uniformly bounded sequence of continuous functions (by Lusin's theorem and dominated
    convergence), so $F(\mu)=\lim_n\int\varphi_n\,d\mu$ is a pointwise limit of Borel measurable
    functions, hence Borel measurable.} Therefore $\omega\mapsto F(S_t(\omega))$ is
    $\sigma(S_t)$-measurable. By $(\star)$, $\mathbb E[\varphi(\theta)\mid\mathcal F_t](\omega)
    =F(S_t(\omega))$ a.s., so $\mathbb E[\varphi(\theta)\mid\mathcal F_t]$ is
    $\sigma(S_t)$-measurable. For any $B\in\sigma(S_t)\subseteq\mathcal F_t$,
    \[
      \mathbb E\bigl[\varphi(\theta)\mathbf 1_B\bigr]
        =\mathbb E\bigl[\mathbb E[\varphi(\theta)\mid\mathcal F_t]\mathbf 1_B\bigr]
        =\mathbb E\bigl[F(S_t)\mathbf 1_B\bigr].
    \]
    Since $F(S_t)$ is $\sigma(S_t)$-measurable, uniqueness of conditional expectation gives
    $\mathbb E[\varphi(\theta)\mid\mathcal F_t]=\mathbb E[\varphi(\theta)\mid S_t]$ a.s. As
    $\varphi$ was arbitrary, conditioning on $S_t$ and conditioning on $\mathcal F_t$ yield the
    same conditional distribution of $\theta$.

    \emph{Part (ii): Induction by $Z_t$.}\quad
    Since $\mathcal F_t=\sigma(Z_t)$ and $S_t$ is $\sigma(Z_t)$-measurable with values in the
    standard Borel space $\Delta(\Theta)$, the Doob--Dynkin lemma (Kallenberg 2002, Lemma~1.14)
    yields a Borel measurable function $f_t:E_t\to\Delta(\Theta)$ such that
    $S_t(\omega)=f_t(Z_t(\omega))$ $\mathbb P$-a.s. Hence $Z_t$ induces the posterior $S_t$
    through the measurable map $f_t$.

    \emph{Part (iii): Martingale property.}\quad
    Fix $s\le t$ and any bounded Borel $\varphi:\Theta\to\mathbb R$. Then
    \[
      \mathbb E\!\left[\int_\Theta\varphi(\theta')\,S_t(\omega)(d\theta')\,\Big|\,\mathcal F_s\right]
        =\mathbb E\bigl[\mathbb E[\varphi(\theta)\mid\mathcal F_t]\mid\mathcal F_s\bigr]
        =\mathbb E[\varphi(\theta)\mid\mathcal F_s]
        =\int_\Theta\varphi(\theta')\,S_s(\omega)(d\theta')\quad\text{a.s.},
    \]
    where the first and last equalities use $(\star)$, and the middle equality uses the tower
    property of conditional expectations together with $\mathcal F_s\subseteq\mathcal F_t$. Since
    $\varphi$ was arbitrary, $\mathbb E[S_t\mid\mathcal F_s]=S_s$ a.s. in $\Delta(\Theta)$.
    Boundedness holds because $S_t(\omega)$ is a probability measure: for every bounded $\varphi$,
    $\bigl|\int_\Theta\varphi\,dS_t\bigr|\le\|\varphi\|_\infty<\infty$.
\end{proof}
